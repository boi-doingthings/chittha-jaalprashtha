Apply Transforms

from torchvision.transforms import ToTensor, Compose
def transform(image):
    a = Compose([ToTensor()])
    return a(image.permute(1,2,0).numpy())
def target_transform(label):
    return 1 if label=='adults' else 0
class CvA_Dataset_with_Transforms(Dataset):
    def __init__(self,labels,files,transform,target_transform):
        self.labels = labels
        self.files = files
        self.transform = transform
        self.target_transform = target_transform
        
    def __len__(self):
        return len(self.files)
    
    def __getitem__(self,idx):
        image = read_image(self.files[idx])
        image = self.transform(image)
        label = self.target_transform(self.labels[idx])
        return image,label

The problem with the our target transform is we don’t get a vector output, that would be useful in training models.
Thus, we try to implement a Generic Transform

from torchvision.transforms import Lambda
import torch
def target_transform(label):
    if label == 'adults':
        y = 1
    else:
        y=0
    return torch.zeros(2).scatter_(dim=0,value=1,index=torch.tensor(y))
